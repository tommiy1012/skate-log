            // ==================== 本地存储 ====================

            function saveToStorage() {

                try {

                    localStorage.setItem(STORAGE_KEY, JSON.stringify(tricks));

                } catch (e) {

                    showToast('⚠️ 存储空间不足，请清理旧数据');

                }

            }

            function loadFromStorage() {

                try {

                    const raw = localStorage.getItem(STORAGE_KEY);

                    if (raw) {

                        const parsed = JSON.parse(raw);

                        if (Array.isArray(parsed)) {

                            tricks = parsed;

                            // 数据迁移/修复

                            tricks.forEach(t => {

                                if (!t.id) t.id = generateId();

                                if (!STATUS_CONFIG[t.status]) t.status = 'untried';

                                if (!CATEGORY_CONFIG[t.category]) t.category = 'other';

                                if (!t.createdAt) t.createdAt = new Date().toISOString();

                            });

                        }

                    }

                } catch (e) {

                    tricks = [];

                    showToast('⚠️ 数据读取失败，已重置');

                }

            }

            function generateId() {

                return 'trick_' + Date.now() + '_' + Math.random().toString(36).slice(2, 8);

            }

            // ==================== Toast ====================

            let toastTimer;

            function showToast(message) {

                clearTimeout(toastTimer);

                toast.textContent = message;

                toast.classList.add('show');

                toastTimer = setTimeout(() => {

                    toast.classList.remove('show');

                }, 2000);

            }

            // ==================== 统计更新 ====================

            function updateStats() {

                const total = tricks.length;

                const mastered = tricks.filter(t => t.status === 'mastered').length;

                const practicing = tricks.filter(t => t.status === 'practicing').length;

                const untried = tricks.filter(t => t.status === 'untried').length;

                // 动画数字

                animateNumber(statTotal, total);

                animateNumber(statMastered, mastered);

                animateNumber(statPracticing, practicing);

                animateNumber(statUntried, untried);

            }

            function animateNumber(element, target) {

                const current = parseInt(element.textContent) || 0;

                if (current === target) return;

                const diff = target - current;

                const steps = Math.min(Math.abs(diff), 20);

                const increment = diff / steps;

                let step = 0;

                const anim = setInterval(() => {

                    step++;

                    if (step >= steps) {

                        element.textContent = target;

                        clearInterval(anim);

                    } else {

                        element.textContent = Math.round(current + increment * step);

                    }

                }, 30);

            }

            // ==================== 渲染列表 ====================

            function getFilteredTricks() {

                if (currentFilter === 'all') return [...tricks];

                if (['mastered', 'practicing', 'untried'].includes(currentFilter)) {

                    return tricks.filter(t => t.status === currentFilter);

                }

                // 按分类筛选

                return tricks.filter(t => t.category === currentFilter);

            }

            function renderTrickList() {

                const filtered = getFilteredTricks();

                if (filtered.length === 0) {

                    trickListContainer.innerHTML = `

                            <div class="empty-state">

                                <span class="emoji">🛹💨</span>

                                <p>${tricks.length === 0 ? '还没有记录任何技巧，<br>快添加你的第一个动作吧！' : '没有匹配的技巧，试试切换筛选条件。'}</p>

                            </div>`;

                    return;

                }

                // 按创建时间倒序排列（最新的在前）

                const sorted = [...filtered].sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));

                trickListContainer.innerHTML = sorted.map(trick => {

                    const catConfig = CATEGORY_CONFIG[trick.category] || CATEGORY_CONFIG['other'];

                    const statusConfig = STATUS_CONFIG[trick.status];

                    const dateStr = new Date(trick.createdAt).toLocaleDateString('zh-CN', {

                        year: 'numeric',

                        month: 'short',

                        day: 'numeric',

                    });

                    return `

                            <div class="trick-card" data-id="${trick.id}">

                                <div class="trick-icon">${catConfig.icon}</div>

                                <div class="trick-info">

                                    <div class="trick-name">${escapeHTML(trick.name)}</div>

                                    <div class="trick-category">${catConfig.label}</div>

                                    <div class="trick-date">📅 ${dateStr}</div>

                                </div>

                                <span class="trick-status-badge ${statusConfig.badgeClass}" 

                                      data-action="cycle-status" 

                                      data-id="${trick.id}"

                                      title="点击切换状态：${statusConfig.label} → ${STATUS_CONFIG[statusConfig.next]?.label || '未尝试'}">

                                    ${statusConfig.icon} ${statusConfig.label}

                                </span>

                                <button class="btn-delete" 

                                        data-action="delete" 

                                        data-id="${trick.id}" 

                                        title="删除此技巧">×</button>

                            </div>`;

                }).join('');

                // 重新统计

                updateStats();

            }

            function escapeHTML(str) {

                const div = document.createElement('div');

                div.textContent = str;

                return div.innerHTML;

            }

            // ==================== 事件委托（列表内的点击） ====================

            trickListContainer.addEventListener('click', function(e) {

                const target = e.target.closest('[data-action]');

                if (!target) return;

                const action = target.getAttribute('data-action');

                const id = target.getAttribute('data-id');

                if (action === 'cycle-status') {

                    cycleTrickStatus(id);

                } else if (action === 'delete') {

                    deleteTrick(id);

                }

            });

            function cycleTrickStatus(id) {

                const trick = tricks.find(t => t.id === id);

                if (!trick) return;

                const oldStatus = trick.status;

                const newStatus = STATUS_CONFIG[oldStatus].next;

                trick.status = newStatus;

                saveToStorage();

                renderTrickList();

                updateFilterButtonStates();

                const statusConfig = STATUS_CONFIG[newStatus];

                showToast(`${STATUS_CONFIG[oldStatus].icon} → ${statusConfig.icon} "${trick.name}" 状态已更新为「${statusConfig.label}」`);

            }

            function deleteTrick(id) {

                const trick = tricks.find(t => t.id === id);

                if (!trick) return;

                const name = trick.name;

                // 简单确认（可自定义为更美观的模态框）

                if (confirm(`确定要删除技巧「${name}」吗？此操作不可恢复。`)) {

                    tricks = tricks.filter(t => t.id !== id);

                    saveToStorage();

                    renderTrickList();

                    updateFilterButtonStates();

                    showToast(`🗑️ 已删除「${name}」`);

                }

            }

            // ==================== 添加技巧 ====================

            function addTrick() {

                const name = trickNameInput.value.trim();

                const category = categorySelect.value;

                if (!name) {

                    showToast('⚠️ 请输入技巧名称');

                    trickNameInput.focus();

                    // 输入框抖动效果

                    trickNameInput.style.animation = 'none';

                    trickNameInput.offsetHeight; // reflow

                    trickNameInput.style.animation = 'shake 0.5s ease';

                    return;

                }

                // 检查重复

                const duplicate = tricks.some(

                    t => t.name.toLowerCase() === name.toLowerCase() && t.category === category

                );

                if (duplicate) {

                    showToast('⚠️ 该分类下已存在同名技巧');

                    trickNameInput.focus();

                    return;

                }

                const newTrick = {

                    id: generateId(),

                    name: name,

                    category: category,

                    status: 'untried',

                    createdAt: new Date().toISOString(),

                };

                tricks.push(newTrick);

                saveToStorage();

                trickNameInput.value = '';

                trickNameInput.focus();

                renderTrickList();

                updateFilterButtonStates();

                showToast(`🛹 已添加「${name}」到 ${CATEGORY_CONFIG[category]?.label || '其他'}！`);

            }

            // 输入框抖动动画

            const shakeKeyframes = `

                    @keyframes shake {

                        0%,100% { transform: translateX(0); }

                        20% { transform: translateX(-8px); }

                        40% { transform: translateX(8px); }

                        60% { transform: translateX(-6px); }

                        80% { transform: translateX(4px); }

                    }`;

            const shakeStyle = document.createElement('style');

            shakeStyle.textContent = shakeKeyframes;

            document.head.appendChild(shakeStyle);

            btnAddTrick.addEventListener('click', addTrick);

            trickNameInput.addEventListener('keydown', function(e) {

                if (e.key === 'Enter') {

                    e.preventDefault();

                    addTrick();

                }

            });

            // ==================== 清空全部 ====================

            btnResetAll.addEventListener('click', function() {

                if (tricks.length === 0) {

                    showToast('📭 记录表已经是空的');

                    return;

                }

                if (confirm(`确定要清空全部 ${tricks.length} 条技巧记录吗？此操作不可恢复！`)) {

                    tricks = [];

                    saveToStorage();

                    renderTrickList();

                    updateFilterButtonStates();

                    currentFilter = 'all';

                    updateFilterButtonStates();

                    showToast('🗑️ 已清空全部记录');

                }

            });

            // ==================== 筛选按钮 ====================

            function updateFilterButtonStates() {

                filterButtons.forEach(btn => {

                    const filterVal = btn.getAttribute('data-filter');

                    if (filterVal === currentFilter) {

                        btn.classList.add('active');

                    } else {

                        btn.classList.remove('active');

                    }

                });

            }

            filterButtons.forEach(btn => {

                btn.addEventListener('click', function() {

                    const filterVal = this.getAttribute('data-filter');

                    currentFilter = filterVal;

                    updateFilterButtonStates();

                    renderTrickList();

                    const label = this.textContent.trim();

                    showToast(`📋 筛选：${label}`);

                });

            });

            function init() {

                if (e.ctrlKey && e.shiftKey && e.key === 'A') {

                    e.preventDefault();

                    trickNameInput.focus();

                    trickNameInput.select();

                if (e.key === 'Escape' && currentFilter !== 'all') {

                    currentFilter = 'all';

                    updateFilterButtonStates();

                    renderTrickList();

                    showToast('📋 已显示全部技巧');

                }

            });

            // ==================== 启动应用 ====================
