        function getEmbedUrl(url) {

            if (!url) return null;

            // YouTube

            let match = url.match(/(?:youtube\.com\/watch\?v=|youtu\.be\/)([a-zA-Z0-9_-]{11})/);

            if (match) return `https://www.youtube.com/embed/${match[1]}`;

            // B站

            match = url.match(/bilibili\.com\/video\/(BV[a-zA-Z0-9]+)/);

            if (match) return `//player.bilibili.com/player.html?bvid=${match[1]}&page=1`;

            // B站短链接

            match = url.match(/b23\.tv\/([a-zA-Z0-9]+)/);

            if (match) return null; // 短链接需要解析，暂返回null让用户用完整链接

            return null;

        }

        function addVideo(trickId) {

            const urlInput = document.getElementById('videoUrlInput');

            const labelInput = document.getElementById('videoLabelInput');

            const url = urlInput.value.trim();

            const label = labelInput.value.trim() || '';

            if (!url) return;

            const videos = loadVideos();

            if (!videos[trickId]) videos[trickId] = [];

            videos[trickId].push({ url, label, addedAt: new Date().toISOString() });

            saveVideos(videos);

            urlInput.value = '';

            labelInput.value = '';

            openTrickDetail(trickId); // 刷新

        }

                saveVideos(videos);

            }

        }

        function renderStats() {

            const logs = loadLogs();

            const statuses = loadStatuses();

            const allTricks = getAllTricks();

            const grid = document.getElementById('statsGrid');

            const detailContainer = document.getElementById('statsDetailContainer');

            const totalAttempts = logs.reduce((sum, l) => sum + (l.attempts || 0), 0);

            const totalLogs = logs.length;

            const uniqueTricksPracticed = new Set(logs.map(l => l.trickId)).size;

            const totalTricks = allTricks.length;

            const masteredCount = Object.values(statuses).filter(s => s === 'mastered').length;

            const practiceCount = Object.values(statuses).filter(s => s === 'practice').length;

            const progressPercent = totalTricks > 0 ? Math.round((masteredCount / totalTricks) * 100) : 0;

            // 连续打卡天数

            const dates = [...new Set(logs.map(l => l.date))].sort((a, b) => new Date(b) - new Date(a));

            let streak = 0;

            if (dates.length > 0) {

                const today = new Date();

                today.setHours(0, 0, 0, 0);

                const yesterday = new Date(today);

                yesterday.setDate(yesterday.getDate() - 1);

                const checkDate = new Date(dates[0]);

                checkDate.setHours(0, 0, 0, 0);

                if (checkDate.getTime() === today.getTime() || checkDate.getTime() === yesterday.getTime()) {

                    streak = 1;

                    for (let i = 1; i < dates.length; i++) {

                        const d1 = new Date(dates[i - 1]);

                        const d2 = new Date(dates[i]);

                        d1.setHours(0, 0, 0, 0);

                        d2.setHours(0, 0, 0, 0);

                        const diff = (d1.getTime() - d2.getTime()) / (1000 * 60 * 60 * 24);

                        if (diff === 1) streak++;

                        else break;

                    }

                }

            }

            grid.innerHTML = `

                <div class="stat-card">

                    <div class="stat-value">${totalLogs}</div>

                    <div class="stat-label">总打卡次数</div>

                </div>

                <div class="stat-card">

                    <div class="stat-value">${totalAttempts}</div>

                    <div class="stat-label">总练习次数</div>

                </div>

                <div class="stat-card">

                    <div class="stat-value">${uniqueTricksPracticed}/${totalTricks}</div>

                    <div class="stat-label">练习动作数</div>

                </div>

                <div class="stat-card">

                    <div class="stat-value">🔥 ${streak}</div>

                    <div class="stat-label">连续打卡天数</div>

                </div>

                <div class="stat-card">

                    <div class="stat-value" style="color:var(--success);">${masteredCount}</div>

                    <div class="stat-label">已掌握动作</div>

                </div>

                <div class="stat-card">

                    <div class="stat-value">${progressPercent}%</div>

                    <div class="stat-label">掌握进度</div>

                    <div class="progress-bar-wrap"><div class="progress-bar-fill" style="width:${progressPercent}%;"></div></div>

                </div>

            `;

            // 每个动作的练习统计

            const trickStats = {};

            for (const log of logs) {

                if (!trickStats[log.trickId]) {

                    trickStats[log.trickId] = { count: 0, totalAttempts: 0, lastDate: log.date };

                }

                trickStats[log.trickId].count++;

                trickStats[log.trickId].totalAttempts += (log.attempts || 0);

                if (log.date > trickStats[log.trickId].lastDate) {

                    trickStats[log.trickId].lastDate = log.date;

                }

            }

            const sortedTricks = Object.entries(trickStats)

                .sort((a, b) => b[1].totalAttempts - a[1].totalAttempts)

                .slice(0, 15);

            if (sortedTricks.length > 0) {

                let detailHtml =

                    '<h3 style="font-size:1rem;font-weight:700;margin:16px 0 10px;">🏆 练习最多的动作（按总次数排序）</h3>';

                for (const [trickId, stats] of sortedTricks) {

                    const trick = getTrickById(trickId);

                    const name = trick ? trick.name : trickId;

                    const catLabel = (trick && trick.category === 'bowl') ? '碗池' : '街市';

                    const status = statuses[trickId] || 'new';

                    const statusEmoji = status === 'mastered' ? '✅' : status === 'practice' ? '🔄' : '⬜';

                    detailHtml += `

                    <div style="display:flex;justify-content:space-between;align-items:center;padding:8px 12px;background:#faf7f3;border-radius:8px;margin-bottom:6px;flex-wrap:wrap;gap:6px;">

                    </div>`;

                }

                detailContainer.innerHTML = detailHtml;

            } else {

                    '<div class="empty-state"><span class="emoji">📊</span><p>开始打卡后这里会显示统计数据</p></div>';

        function importData(event) {

            const btn = document.querySelector(`[data-tab="${tabName}"]`);

            const content = document.getElementById(tabName);

            if (btn) btn.classList.add('active');

            if (content) content.classList.add('active');

            if (tabName === 'tab-log') renderLogs();

            if (tabName === 'tab-stats') renderStats();
